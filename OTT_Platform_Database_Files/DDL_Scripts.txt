CREATE TABLE Subscription_Plans (  subscription_plan_id INT PRIMARY KEY,  plan_name VARCHAR(50),  price DECIMAL(10,2),  duration INT,  max_quality VARCHAR(10),  max_number_of_screens INT,  max_concurrent_streams INT);CREATE TABLE Videos (  video_id SERIAL PRIMARY KEY,  title VARCHAR(255),  language CHAR(2),  description TEXT,  views INT,  subtitles VARCHAR(255),  duration FLOAT,  release_year INT,  rating VARCHAR(10),  imdb_rating FLOAT,  video_type_id INT);CREATE TABLE users (user_id INT PRIMARY KEY,username VARCHAR(30),email VARCHAR(50),subscription_plan_id INT REFERENCES subscription_plans(subscription_plan_id),prev_watched_video_id  INT  REFERENCES videos(video_id));CREATE TABLE sub_history (subscription_plan_id
